00159 ParticleData * ParticleDataTable::DealWithUnknownID( ParticleID & key ) const
00160 {
00161   // call special lookup function and return that value
00162   CommonParticleData * cpd = itsProcessUnknownID->callProcessUnknownID( key, *this );
00163   if( !cpd ) return 0;
00164   // have the CPD, now create the PD
00165   CPDID cid = addParticleData(*cpd);
00166   // a ParticleData will ALWAYS have a DecayData, which can be empty
00167   // The first DDID in the list will be an empty DecayData 
00168   // (see the convertTemporaryMap code)
00169   DDID edid = itsDDlist.begin();
00170   // there is no decay information for this ParticleData
00171   ParticleData * pd = new ParticleData( cid, edid );
00172   addParticle( *pd );
00173   ParticleID id = pd->ID();
00174   // cleanup of temporarys
00175   delete pd;
00176   delete cpd;
00177   // find the real pd
00178   return & itsMap.find( id )->second;
00179 }
