00505             charge = -charge; 
00506         }

00508         charge = ch100[sid-1];

00510         if(ida==1000034 || ida==1000052) { charge = 0; }
00511         if(ida==1000053 || ida==1000054) { charge = 0; }

00514         return 0;

00516             if( q2 == 3 || q2 == 5 ) {

00539     }
